	name:RPG Clicker
	by:Nozzles123
	desc:An incremental rpg game.
	created:26/11/2019
	updated:26/11/2019
	version:0.0.2

Settings
  background:https://raw-nozzite.github.io/RPGclicker/Images/Wood.jpg
  building cost increase:115%
  building cost refund:50%
  
Layout
	*resourceButtons
		contains:Buttons
		header:These are the buttons

Buttons
	*attackButton
		name:Gain Attack
		desc:Click this to gain attack points.
		on click:anim icon wobble
		on click:yield 1 attack
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red
		
	*defenseButton
		name:Gain Defense
		desc:Click this to gain defense points.
		on click:anim icon wobble
		on click:yield 1 defense
		class:bigButton hasFlares
		icon class:shadowed
		tooltip origin:bottom
		tooltip class:red

Resources
    *attack|attack
        name:Attack|Attack
        desc:This is how many attack points you have.
        class:noBackground
        show earned
	
    *defense|defense
        name:Defense|Defense
        desc:This is how many defense point you have.
        class:noBackground
        hidden when 0
